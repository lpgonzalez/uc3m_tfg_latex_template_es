% --- Salida PDF/A (para el depósito en e-Archivo) -------------------
% Cambia false por true para generar un PDF/A-2b. Para la versión final, oculta
% el capítulo de demostración con \showDemoChapter=false (es contenido de
% ejemplo) y valida el PDF con veraPDF. DEBE estar al principio del todo:
% \DocumentMetadata tiene que ser el primer comando, incluso antes de
% \documentclass; por eso este interruptor vive aquí, no en tfg_vars.tex.
\def\pdfaOutput{false}
\def\tfgpdfayes{true}%
\ifx\pdfaOutput\tfgpdfayes
  \DocumentMetadata{pdfstandard=A-2b,pdfversion=1.7}% PDF/A-2b (PDF 1.7)
\else
  \pdfvariable minorversion=7 % versión normal: PDF 1.7 moderno (LuaTeX usa 1.5)
\fi

% Tus datos y ajustes.
\input{tfg_vars.tex}
% Metadatos de la plantilla (versión y procedencia); los mantiene quien
% actualiza la plantilla, no hace falta tocarlo para tu TFG.
\input{tfg_template_info.tex}

\documentclass[twoside,12pt]{book}

% Estilo TFG UC3M: paquetes, maquetación, tipografía y metadatos del PDF.
% Mantén esto como primer paquete; carga todo lo demás.
\usepackage{tfg_uc3m}

% Bibliografía: estilo elegido por \bibStyle en tfg_vars.tex (apa / ieee).
\custombib{\bibStyle}
\addbibresource{bibliography/bibliography.bib}

% Entradas del glosario y de los acrónimos.
\loadglsentries{others/glossary}

\begin{document}
\selectlanguage{spanish}

% =====================================================================
%  PÁGINAS PRELIMINARES
% =====================================================================
% Numeración romana desde el inicio, para que la portada/agradecimientos y
% el cuerpo no compartan anclas de página (evita avisos de destino duplicado).
\pagenumbering{Roman}
\pagestyle{empty}

% --- Portada ---------------------------------------------------------
\makecover
\cleardoublepage

% --- Agradecimientos / dedicatoria -----------------------------------
\input{others/acks}
\clearpage

\pagestyle{plain}

% --- Resumen (español) y Abstract (inglés) ---------------------------
% La UC3M suele exigir ambos resúmenes. Elimina el que no necesites.
\input{others/resumen}
\clearpage
\input{others/abstract}
\clearpage

% --- Índices de contenidos, figuras y tablas -------------------------
% Encadenados SIN páginas en blanco entre ellos: dentro del grupo, el
% \cleardoublepage que insertan \tableofcontents/\listof... (clase book,
% openright) se degrada a \clearpage, de modo que cada índice empieza justo
% tras el anterior en vez de forzar página impar.
\begingroup
\makeatletter\@openrightfalse\makeatother
\let\cleardoublepage\clearpage
\tableofcontents
\listoffigures
\listoftables
\endgroup
\clearpage

% =====================================================================
%  CUERPO
% =====================================================================
\pagenumbering{arabic}              % Numeración arábiga desde el primer capítulo
\pagestyle{tfgbody}

% Capítulo de demostración con las principales funciones de la plantilla.
% Se muestra solo si \showDemoChapter = true (en tfg_vars.tex); ponlo a false
% para la versión final / PDF/A. También puedes borrar chapters/00_examples.tex.
\ifdefstring{\showDemoChapter}{true}{\input{chapters/00_examples}}{}

\input{chapters/01_introduction}
\input{chapters/02_soa}
\input{chapters/03_method}
\input{chapters/04_validation}
\input{chapters/05_resultsanddiscussion}
\input{chapters/06_projectmanagement}
\input{chapters/07_conclusions}
\input{chapters/08_futurelines}

% =====================================================================
%  PÁGINAS FINALES
% =====================================================================
% --- Acrónimos y glosario --------------------------------------------
\printglossary[type=\acronymtype,title=Acrónimos]
\printglossary[type=main]

% --- Bibliografía ----------------------------------------------------
\printbibliography[heading=bibintoc]

% --- Anexos / apéndices ----------------------------------------------
\input{chapters/09_annexes}

% --- Anexo obligatorio: declaración de uso de IAG --------------------
\input{others/ai_declaration}
\cleardoublepage

\end{document}
